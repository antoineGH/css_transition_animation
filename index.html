CSS Transitions / Animations

Transition 1

Ease-In-Out - Left to Right - 1s

.child {
  transition: 1s ease-in-out;
}

.parent:hover .child {
  transform: translateX(100%);
}

Transition 2

Ease-In-Out - Left to Right - 1s - Color

.child {
  transition: 1s ease-in-out;
}

.parent:hover .child {
  transform: translateX(100%);
  background-color: green;
}

Transition 3

Linear - Left to Right - 1s

}

.parent:hover .child {

}

Transition 4

Custom Cubic Bezier - Left to Right - 1s

.child {

Animation 1

Ease In Forward - Left to Right - 1s

.child {
  transition: transform 1s ease-in;
}

.parent:hover .child {
  animation: left-to-right 1s ease-in forwards;
}

@keyframes left-to-right {
  100% {
    transform: translateX(100%);
  }
}

Animation 2

Infinite Reverse - Cover

.child {
  animation: left-to-right-step 1s ease-in forwards infinite alternate;
}

.parent:hover .child {
  animation-play-state: paused;
}

@keyframes left-to-right-step {
  0% {
    transform: translateX(0%);
  }

  33% {
    transform: translateY(100%);
  }

  66% {
    transform: translateY(100%) translateX(100%);
  }

  100% {
    transform: translateX(100%);
  }
}

  }
}
